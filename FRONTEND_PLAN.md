## 3. PAGES BREAKDOWN

### `/login` and `/register`
- shadcn `Card` centered on page
- `Form` + `Input` + `Button` from shadcn
- React Hook Form + Zod validation
- On success → store JWT in Zustand + redirect to `/search`

### `/search` (Main Page)
- Large `Textarea` for NL prompt
- **"Search"** `Button` with loading spinner
- Credit cost warning: `"This search will use 1 credit (You have: X)"`
- On result:
  - **FilterPreview card** shows what AI parsed (collapsible)
  - **SearchResultsTable** with leads
  - Export to CSV button
- Error state via `Sonner` toast

### `/history`
- List of past searches as `Card` components
- Each card shows: prompt snippet, date, result count, credits used
- Click → goes to `/history/[id]`

### `/history/[id]`
- Shows full leads table for that saved search
- Re-export option

### `/credits`
- Current balance in large typography
- Transaction history table (shadcn `DataTable`)

## 4. KEY COMPONENT — SearchBox.tsx

```tsx
'use client'
import { useState } from 'react'
import { Button } from '@/components/ui/button'
import { Textarea } from '@/components/ui/textarea'
import { Card, CardContent, CardHeader, CardTitle } from '@/components/ui/card'
import { Badge } from '@/components/ui/badge'
import { Search, Loader2 } from 'lucide-react'
import { useSearch } from '@/hooks/useSearch'
import { useCredits } from '@/hooks/useCredits'

export function SearchBox() {
  const [prompt, setPrompt] = useState('')
  const { mutate: runSearch, isPending, data, error } = useSearch()
  const { data: credits } = useCredits()

  const handleSearch = () => {
    if (!prompt.trim()) return
    runSearch({ prompt, page: 1 })
  }

  return (
    <Card className="w-full max-w-3xl mx-auto">
      <CardHeader>
        <CardTitle className="flex items-center justify-between">
          Find Leads with AI
          <Badge variant="outline">
            {credits?.balance ?? 0} credits remaining
          </Badge>
        </CardTitle>
      </CardHeader>
      <CardContent className="space-y-4">
        <Textarea
          placeholder='e.g. "Find 5 CTOs at fintech startups in Canada with 50-500 employees"'
          value={prompt}
          onChange={(e) => setPrompt(e.target.value)}
          rows={3}
          className="resize-none"
        />
        <div className="flex items-center justify-between">
          <p className="text-sm text-muted-foreground">
            This search will use <strong>1 credit</strong>
          </p>
          <Button onClick={handleSearch} disabled={isPending || !prompt.trim()}>
            {isPending ? (
              <><Loader2 className="mr-2 h-4 w-4 animate-spin" /> Searching...</>
            ) : (
              <><Search className="mr-2 h-4 w-4" /> Search</>
            )}
          </Button>
        </div>
      </CardContent>
    </Card>
  )
}
```

## 5. KEY HOOK — useSearch.ts

```typescript
import { useMutation, useQueryClient } from '@tanstack/react-query'
import { api } from '@/lib/api'
import { toast } from 'sonner'

interface SearchPayload {
  prompt: string
  page: number
}

export function useSearch() {
  const queryClient = useQueryClient()

  return useMutation({
    mutationFn: async (payload: SearchPayload) => {
      const { data } = await api.post('/prospects/search', payload)
      return data
    },
    onSuccess: (data) => {
      toast.success(`Found ${data.leads.length} leads`)
      // Invalidate credits and history
      queryClient.invalidateQueries({ queryKey: ['credits'] })
      queryClient.invalidateQueries({ queryKey: ['searches'] })
    },
    onError: (error: any) => {
      const msg = error?.response?.data?.error || 'Search failed'
      toast.error(msg)
    }
  })
}
```

## 7. AXIOS INSTANCE — lib/api.ts

```typescript
import axios from 'axios'

export const api = axios.create({
  baseURL: process.env.NEXT_PUBLIC_API_URL || 'http://localhost:5000/api/v1',
  withCredentials: true
})

// Attach JWT from localStorage on every request
api.interceptors.request.use((config) => {
  const token = localStorage.getItem('token')
  if (token) config.headers.Authorization = `Bearer ${token}`
  return config
})

// Auto logout on 401
api.interceptors.response.use(
  (res) => res,
  (error) => {
    if (error.response?.status === 401) {
      localStorage.removeItem('token')
      window.location.href = '/login'
    }
    return Promise.reject(error)
  }
)
```

## 9. ZUSTAND STORE — store/authStore.ts

```typescript
import { create } from 'zustand'
import { persist } from 'zustand/middleware'

interface User {
  id: string
  email: string
  credits: number
}

interface AuthStore {
  user: User | null
  token: string | null
  setAuth: (user: User, token: string) => void
  logout: () => void
}

export const useAuthStore = create<AuthStore>()(
  persist(
    (set) => ({
      user: null,
      token: null,
      setAuth: (user, token) => {
        localStorage.setItem('token', token)
        set({ user, token })
      },
      logout: () => {
        localStorage.removeItem('token')
        set({ user: null, token: null })
      }
    }),
    { name: 'auth-storage' }
  )
)
```

## 13. ROUTE PROTECTION — middleware.ts

```typescript
// middleware.ts (Next.js root)
import { NextResponse } from 'next/server'
import type { NextRequest } from 'next/server'

const PUBLIC_PATHS = ['/login', '/register']

export function middleware(request: NextRequest) {
  const token = request.cookies.get('token')?.value
  const isPublic = PUBLIC_PATHS.some(p => request.nextUrl.pathname.startsWith(p))

  if (!token && !isPublic) {
    return NextResponse.redirect(new URL('/login', request.url))
  }
  if (token && isPublic) {
    return NextResponse.redirect(new URL('/search', request.url))
  }

  return NextResponse.next()
}

export const config = {
  matcher: ['/((?!api|_next/static|_next/image|favicon.ico).*)']
}
```
